local function XZVXSM_fake_script() -- less.LocalScript 
	local script = Instance.new('LocalScript', less)

	local newVar = Instance.new("StringValue")
	newVar.Parent = script.Parent.Parent
	newVar.Name = "FrameV"
	newVar.Value = "WelcomeMenu"
	local toggle = false
	
	local function hideFrames()
		for _,frame in pairs(script.Parent.Parent:GetChildren()) do
			if frame:IsA("Frame") then
				frame.Visible = false
			end
		end
	end
	
	script.Parent.MouseButton1Click:Connect(function()
		if toggle == false then
			toggle = true
			script.Parent.Text = "+"
			hideFrames()
		else
			toggle = false
			script.Parent.Text = "-"
			for _,showSpeceficFrame in pairs(script.Parent.Parent:GetChildren()) do
				if showSpeceficFrame.Name == newVar.Value then
					showSpeceficFrame.Visible = true
					script.Parent.Parent.ButtonBar.Visible = true
				end
			end
		end
	end)
end

local function WMCTZT_fake_script() -- TextButton.LocalScript 
	local script = Instance.new('LocalScript', TextButton)

	local Players = game:GetService("Players")
	local player = Players.LocalPlayer
	local PlayerGui = player:WaitForChild("PlayerGui")
	local stats = player:WaitForChild("stats")  -- your stats folder
	
	-- Keep track of the loop
	local xpUpdating = false
	
	-- Reference for the toggle button text
	local toggleButtonText = "BETTER XP"
	local toggleButton = script.Parent
	
	-- Function to toggle XP display
	local function toggleXP()
		local existingGui = PlayerGui:FindFirstChild("XPGui")
		if existingGui then
			existingGui:Destroy()
			xpUpdating = false
			toggleButton.Text = toggleButtonText  -- reset button text
			return
		end
	
		-- Create ScreenGui for XP label
		local screenGui = Instance.new("ScreenGui")
		screenGui.Name = "XPGui"
		screenGui.Parent = PlayerGui
	
		local xpLabel = Instance.new("TextLabel")
		xpLabel.Size = UDim2.new(0, 400, 0, 20)
		xpLabel.Position = UDim2.new(0.5, -200, -0.06, 0)
		xpLabel.BackgroundTransparency = 1
		xpLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
		xpLabel.TextScaled = true
		xpLabel.Font = Enum.Font.Highway
		xpLabel.Parent = screenGui
	
		-- Change button text to "CLOSE"
		toggleButton.Text = "CLOSE"
	
		-- Start updating loop
		xpUpdating = true
		coroutine.wrap(function()
			local getLvlXp = require(game:GetService("ReplicatedStorage").modules.essentials).get_lvl_xp
			while xpUpdating and xpLabel.Parent do
				local currentXP, nextXP = getLvlXp(
					stats:WaitForChild("role_progress").Value,
					stats:WaitForChild("role").Value
				)
				xpLabel.Text = currentXP.."/"..nextXP
				task.wait(0.5)  -- update every 0.5 seconds
			end
		end)()
	end
	
	-- Connect button click
	toggleButton.MouseButton1Click:Connect(toggleXP)
end
coroutine.wrap(WMCTZT_fake_script)()

		conn = RunService.RenderStepped:Connect(function()
			if not toggle or not player.Parent or not player.Character or not player.Character:FindFirstChild("Head") then
				conn:Disconnect()
				if billboard then billboard:Destroy() end
				billboards[player] = nil
				return
			end
	
			local stats = player:FindFirstChild("stats")
			if stats and stats:FindFirstChild("metal") then
				textLabel.Text = tostring(stats.metal.Value)
			else
				textLabel.Text = "0"
			end
		end)
	end
	
	-- Handle character respawn
	local function onCharacterAdded(character, player)
		character:WaitForChild("Head")
		if toggle then
			createBillboard(player)
		end
	end

	ToggleButton.MouseButton1Click:Connect(function()
